import { useState } from 'react'

export function Counter() {
  const [count, setCount] = useState(0)
  return (
    <button 
      onClick={() => setCount(count + 1)}
      style={{
        padding: '10px 20px',
        fontSize: '16px',
        background: 'var(--color-primary)',
        color: 'white',
        border: 'none',
        borderRadius: '4px',
        cursor: 'pointer'
      }}
    >
      Clicked {count} times
    </button>
  )
}
